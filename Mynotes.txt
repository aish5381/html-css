select
    name, salary, commission,
    (salary + ((salary*commission) / 100)) as "total_salary" 
from myemp;



SELECT CONCAT(TIMESTAMPDIFF(YEAR, hire_date, CURDATE()), '  Years  ',
TIMESTAMPDIFF(MONTH, hire_date, CURDATE())%12, '  Month  ',
TIMESTAMPDIFF(DAY, (hire_date + INTERVAL TIMESTAMPDIFF(YEAR, hire_date,CURDATE()) YEAR 
+ INTERVAL TIMESTAMPDIFF(MONTH, hire_date, CURDATE())%12 MONTH), CURDATE()), '  Days') 
AS `Duration` FROM employees;
